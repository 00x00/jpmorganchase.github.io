Tags: advanced

This is to verify the basic functionalities when certain block height is reached. The purpose is to increase the possibility for the network to accept transactions during edge scenarios like: proposer rotation.

Using the following smart contracts:

pragma solidity ^0.4.21;


contract SimpleStorage {
    uint private storedData;


    constructor(uint initVal) public {
        storedData = initVal;
    }


    function set(uint x) public {
        storedData = x;
    }


    function get() public view returns (uint retVal) {
        return storedData;
    }
}


pragma solidity ^0.4.21;


contract ClientReceipt {
    event Deposit(
        address indexed _from,
        bytes32 indexed _id,
        uint _value
    );


    function deposit(bytes32 _id) public payable {
        emit Deposit(msg.sender, _id, msg.value);
    }
}
